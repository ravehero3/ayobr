    <script>
        function switchView() {
            const select = document.getElementById('versionSelect');
            const mainContent = document.getElementById('mainContent');
            
            if (select.value === 'grid') {
                mainContent.className = 'grid-view';
                mainContent.innerHTML = document.querySelector('body').innerHTML.match(/<div id="mainContent" class="grid-view">([\s\S]*?)<\/div>/)[1];
            } else {
                mainContent.className = 'single-view';
                const versionNumber = select.value;
                const versionElement = document.querySelector(`.version-${versionNumber}`).outerHTML;
                mainContent.innerHTML = versionElement;
            }
        }

        function closeProfile() {
            alert('Profile window closed');
        }

        function openModal() {
            document.getElementById('profilePictureModal').classList.add('active');
        }

        function closeModal() {
            document.getElementById('profilePictureModal').classList.remove('active');
        }

        function handleImageUpload(event) {
            const file = event.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = function(e) {
                    const profilePictures = document.querySelectorAll('.profile-picture');
                    profilePictures.forEach(picture => {
                        picture.style.backgroundImage = `url(${e.target.result})`;
                        picture.style.backgroundSize = 'cover';
                        picture.style.backgroundPosition = 'center';
                        picture.innerHTML = '';
                    });
                };
                reader.readAsDataURL(file);
                closeModal();
            }
        }

        function saveProfile(version) {
            const name = document.getElementById(`userName${version}`).value;
            const email = document.getElementById(`userEmail${version}`).value;
            
            if (name.trim() === '') {
                alert('Please enter your name');
                return;
            }
            
            const button = event.target;
            button.textContent = 'Saving...';
            button.style.background = 'linear-gradient(145deg, rgba(34, 197, 94, 0.2) 0%, rgba(0, 0, 0, 1) 100%)';
            
            setTimeout(() => {
                button.textContent = 'Saved!';
                setTimeout(() => {
                    button.textContent = 'Save Changes';
                    button.style.background = 'linear-gradient(145deg, #0a0a0a 0%, #000000 100%)';
                }, 1000);
            }, 1000);
        }

        function cancelChanges(version) {
            document.getElementById(`userName${version}`).value = '';
            document.getElementById(`userEmail${version}`).value = '';
            
            const profilePicture = document.querySelector(`.version-${version} .profile-picture`);
            profilePicture.style.backgroundImage = '';
            profilePicture.innerHTML = '<div class="profile-picture-icon">👤</div>';
        }

        // Close modal when clicking outside
        window.onclick = function(event) {
            const modal = document.getElementById('profilePictureModal');
            if (event.target === modal) {
                closeModal();
            }
        }
