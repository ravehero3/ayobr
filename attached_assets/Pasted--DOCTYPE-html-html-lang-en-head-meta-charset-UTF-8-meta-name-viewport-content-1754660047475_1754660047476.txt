    <script>
        // Mouse-reactive particle system
        document.addEventListener('mousemove', function(e) {
            const containers = document.querySelectorAll('.video-container');
            
            containers.forEach(container => {
                const rect = container.getBoundingClientRect();
                const particles = container.querySelectorAll('.video-particle');
                
                // Calculate mouse position relative to container
                const mouseX = e.clientX - rect.left;
                const mouseY = e.clientY - rect.top;
                
                // Check if mouse is over the container
                const isHovering = mouseX >= 0 && mouseX <= rect.width && 
                                 mouseY >= 0 && mouseY <= rect.height;
                
                particles.forEach((particle, index) => {
                    if (isHovering) {
                        // Calculate distance from mouse to particle
                        const particleRect = particle.getBoundingClientRect();
                        const containerParticleX = particleRect.left - rect.left + particleRect.width/2;
                        const containerParticleY = particleRect.top - rect.top + particleRect.height/2;
                        
                        const deltaX = mouseX - containerParticleX;
                        const deltaY = mouseY - containerParticleY;
                        const distance = Math.sqrt(deltaX * deltaX + deltaY * deltaY);
                        
                        // Maximum influence distance
                        const maxDistance = 150;
                        
                        if (distance < maxDistance) {
                            const influence = 1 - (distance / maxDistance);
                            const moveX = -deltaX * influence * 0.3;
                            const moveY = -deltaY * influence * 0.3;
                            const scale = 1 + influence * 0.5;
                            
                            particle.style.opacity = Math.min(1, influence * 2);
                            particle.style.transform = `translate(${moveX}px, ${moveY}px) scale(${scale})`;
                            particle.style.boxShadow = `0 0 ${influence * 15}px rgba(59, 130, 246, ${influence * 0.8})`;
                        } else {
                            particle.style.opacity = 0.3;
                            particle.style.transform = 'translate(0, 0) scale(1)';
                            particle.style.boxShadow = 'none';
                        }
                    } else {
                        // Reset particles when not hovering
                        particle.style.opacity = 0;
                        particle.style.transform = 'translate(0, 0) scale(1)';
                        particle.style.boxShadow = 'none';
                    }
                });
            });
        });

        // Enhanced hover effects for video containers
        const videoContainers = document.querySelectorAll('.video-container');
        
        videoContainers.forEach(container => {
            container.addEventListener('mouseenter', function() {
                const particles = this.querySelectorAll('.video-particle');
                particles.forEach(particle => {
                    particle.style.opacity = '0.6';
                });
            });
            
            container.addEventListener('mouseleave', function() {
                const particles = this.querySelectorAll('.video-particle');
                particles.forEach(particle => {
                    particle.style.opacity = '0';
                    particle.style.transform = 'translate(0, 0) scale(1)';
                    particle.style.boxShadow = 'none';
                });
            });
        });
